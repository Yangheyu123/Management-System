# 03 - 数据库设计文档

## 1. 数据库总览

- 数据库名：`property`（推荐，对应原 demo 库的升级）
- 字符集：`utf8mb4`，排序规则：`utf8mb4_general_ci`
- 引擎：`InnoDB`
- 命名规范：表名小写蛇形（`sys_user`）；主键统一 `id BIGINT AUTO_INCREMENT`；审计字段统一 `create_time` / `update_time` / `create_by` / `update_by` / `deleted`

> 本文档 SQL 兼容原 `db/init.sql` 的 demo 库结构，并将其扩展为完整的物业系统。可直接执行文末《附录：完整建库 SQL》。

## 7. 附录：完整建库 SQL（可执行）

```sql
-- ============================================================
-- 小区物业管理系统 建库脚本
-- 执行：mysql -u root -p < init_property.sql
-- 或在 Navicat 中全选运行
-- ============================================================

CREATE DATABASE IF NOT EXISTS property DEFAULT CHARACTER SET utf8mb4;
USE property;

-- ---------- 1. sys_user 系统用户 ----------
DROP TABLE IF EXISTS sys_user;
CREATE TABLE sys_user (
  id              BIGINT       PRIMARY KEY AUTO_INCREMENT,
  username        VARCHAR(50)  NOT NULL,
  password        VARCHAR(100) NOT NULL,
  real_name       VARCHAR(50)  NOT NULL,
  phone           VARCHAR(20),
  email           VARCHAR(100),
  gender          TINYINT      DEFAULT 0,
  avatar          VARCHAR(255),
  user_type       TINYINT      NOT NULL DEFAULT 1,
  community_id    BIGINT,
  status          TINYINT      NOT NULL DEFAULT 1,
  last_login_time DATETIME,
  create_time     DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP,
  update_time     DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  create_by       VARCHAR(50),
  update_by       VARCHAR(50),
  deleted         TINYINT      NOT NULL DEFAULT 0,
  UNIQUE KEY uk_username (username),
  KEY idx_community (community_id),
  KEY idx_phone (phone)
) ENGINE=InnoDB COMMENT='系统用户';

-- ---------- 2. sys_role 角色 ----------
DROP TABLE IF EXISTS sys_role;
CREATE TABLE sys_role (
  id          BIGINT      PRIMARY KEY AUTO_INCREMENT,
  role_name   VARCHAR(50) NOT NULL,
  role_code   VARCHAR(50) NOT NULL,
  description VARCHAR(200),
  status      TINYINT     NOT NULL DEFAULT 1,
  create_time DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP,
  update_time DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  create_by   VARCHAR(50),
  update_by   VARCHAR(50),
  deleted     TINYINT     NOT NULL DEFAULT 0,
  UNIQUE KEY uk_role_code (role_code)
) ENGINE=InnoDB COMMENT='角色';

-- ---------- 3. sys_permission 权限 ----------
DROP TABLE IF EXISTS sys_permission;
CREATE TABLE sys_permission (
  id          BIGINT       PRIMARY KEY AUTO_INCREMENT,
  parent_id   BIGINT       NOT NULL DEFAULT 0,
```
